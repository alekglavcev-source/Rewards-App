<script>
const { useState, useEffect } = React;

/* ================= ADMIN DASHBOARD ================= */

function AdminPanel({user}) {
  const [users,setUsers] = useState([]);
  const [name,setName] = useState('');
  const [amount,setAmount] = useState(0);

  async function loadUsers(){
    const res = await fetch('/api/leaderboard');
    const data = await res.json();
    setUsers(data);
  }

  async function giveCoins(){
    await fetch('/api/admin/give',{
      method:'POST',
      headers:{'Content-Type':'application/json'},
      credentials:'include',
      body:JSON.stringify({username:name,amount:Number(amount)})
    });
    loadUsers();
  }

  useEffect(()=>{loadUsers();},[]);

  return (
    <div className="mt-10 p-4 bg-gray-800 rounded-xl">
      <h2 className="text-xl mb-4">Admin Dashboard</h2>

      <div className="mb-4">
        <input placeholder="Username" onChange={e=>setName(e.target.value)} className="p-2 text-black mr-2" />
        <input placeholder="Coins" type="number" onChange={e=>setAmount(e.target.value)} className="p-2 text-black mr-2" />
        <button onClick={giveCoins} className="bg-red-500 p-2 rounded">Give Coins</button>
      </div>

      <h3 className="mb-2">Top Users</h3>
      <ul>
        {users.map((u,i)=>(
          <li key={i} className="border-b border-gray-600 p-1">
            {u.username} - {u.coins} coins
          </li>
        ))}
      </ul>
    </div>
  );
}

/* ================= MAIN APP ================= */

function App() {
  const [user, setUser] = useState(null);

  async function login(u,p) {
    await fetch('/api/login', {
      method:'POST', 
      headers:{'Content-Type':'application/json'},
      body:JSON.stringify({username:u,password:p}),
      credentials:'include'
    });
    load();
  }

  async function load() {
    const res = await fetch('/api/me',{credentials:'include'});
    const data = await res.json();
    setUser(data);
  }

  async function daily() {
    await fetch('/api/daily',{method:'POST',credentials:'include'});
    load();
  }

  async function task() {
    await fetch('/api/task',{method:'POST',credentials:'include'});
    load();
  }

  useEffect(() => {
    load();
  }, []);

  if (!user) return (
    <div className="p-10">
      <h1 className="text-2xl mb-4">Login</h1>
      <button onClick={()=>login('godking3000','arhnhb83bkana9n4')} className="bg-blue-500 p-2 rounded">Login as Admin</button>
    </div>
  );

  return (
    <div className="p-10">
      <h1 className="text-2xl">Welcome {user.username}</h1>
      <p>Coins: {user.coins}</p>
      <p>Referral Code: {user.referralCode}</p>

      <button onClick={daily} className="bg-green-500 p-2 m-2 rounded">Daily Claim</button>
      <button onClick={task} className="bg-purple-500 p-2 m-2 rounded">Complete Task</button>

      {user.username === 'godking3000' && <AdminPanel user={user} />}
    </div>
  );
}

ReactDOM.createRoot(document.getElementById('app')).render(<App/>);
</script>
